python datadasher.py "movies.csv" [OPTIONS]

FILTERS:
  --genre GENRE        Animation, Action, Comedy, etc.
  --season SEASON      Winter, Spring, Summer, Fall
  --years YYYY or YYYY-YYYY    1994 or 1990-1999
  --rating RATING      G, PG, PG-13, R, NC-17
  --min-gross MILLIONS  Minimum box office in millions

SORTING:
  --sort gross         By box office (default: inflation)
  --sort inflation     By inflation-adjusted gross
  --sort date          By release date
  --sort title         By movie title
  --sort year          By release year
  --asc                Sort ascending

OUTPUT:
  --limit N            Show N results (default: 20)
  --csv                Output as CSV format
  --info               Show database statistics

EXAMPLES

# Summer blockbusters over $100M
python datadasher.py "disney movie total gross.csv" --season Summer --min-gross 100 --sort gross

# 90s action movies
python datadasher.py "warner bros.csv" --genre Action --years 1990-1999 --sort gross

# PG-13 movies from 2000-2010
python datadasher.py "disney movie total gross.csv" --rating PG-13 --years 2000-2010 --sort year

# Top 5 highest grossing animated films
python datadasher.py "disney movie total gross.csv" --genre Animation --sort gross --limit 5

# Export all horror movies to CSV
python datadasher.py "warner bros.csv" --genre Horror --csv > horror_movies.csv

# Christmas movies (Winter season)
python datadasher.py "disney movie total gross.csv" --season Winter --sort gross
